\subsection{Output parsing and quality control}
\label{app:parsing}
\textbf{Parsing.} Each response is parsed as JSON. If parsing fails, the request is re-sent once with an appended instruction: \texttt{"Respond with valid JSON only."}~Responses that fail twice ($<$2\% of calls) are discarded. All probabilities are clipped to $[0,1]$; missing keys are filled with 0.3 (the uncertain default).

\noindent\textbf{Cross-teacher consistency.} A random 5\% overlap (${\sim}$1{,}000 posts) is labeled by both teachers. Per-concept Pearson $r$ averages 0.78 (range: 0.71--0.86); highest: Hopelessness ($r{=}0.86$), Self-blame ($r{=}0.83$); lowest: All-or-nothing ($r{=}0.71$).

\noindent\textbf{Intra-teacher stability.} 500 posts are re-queried after 48\,h. Mean absolute deviation per concept is 0.04 (max 0.08), confirming low stochasticity at temperature 0.0.

\section{Hashing and Salt Mechanics}
\label{app:hashing}

\subsection{Pseudocode (client-side hashing)}
\begin{algorithm}[h]
\caption{Salted signed feature hashing for char $n$-grams}
\label{alg:hash}
\begin{algorithmic}[1]
\Require Post text $x'$, salt $\sigma$, dims $m$, $n\_set=\{3,4,5\}$
\State $G \gets$ extract\_char\_ngrams$(x', n\_set)$
\State $\phi \gets \mathbf{0} \in \mathbb{R}^{m}$
\For{$g \in G$}
  \State $j \gets \text{IndexHash}(g, \sigma) \bmod m$
  \State $s \gets \text{SignHash}(g, \sigma) \in \{-1,+1\}$
  \State $\phi[j] \gets \phi[j] + s$
\EndFor
\State \Return $\phi$
\end{algorithmic}
\end{algorithm}

\subsection{Collision notes}
Let $N$ be the number of distinct $n$‑grams; the expected occupied bins with uniform hashing is
$\mathbb{E}[\text{occupied}]=m\big(1 - e^{-N/m}\big)$.  
In practice (train splits):
\begin{itemize}[leftmargin=1.2em,itemsep=1pt]
\item WU3D: $N{\approx}1.8{\times}10^{5}$, $m{=}2^{18}$ $\Rightarrow$ occupied $\approx 1.30{\times}10^{5}$; collision ratio $\approx 27.8\%$.
\item SWDD: $N{\approx}2.4{\times}10^{5}$, $m{=}2^{18}$ $\Rightarrow$ occupied $\approx 1.61{\times}10^{5}$; collision ratio $\approx 32.9\%$.
\end{itemize}
Empirically, sign hashing mitigates bias from systematic collisions. Increasing $m$ to $2^{20}$ reduces collisions by ${\sim}40\%$ with modest memory/time cost.

\paragraph{Rotation cadence vs.\ longitudinal utility.}
Salt rotation has two effects with opposite sign for utility: (i) it breaks cross-deployment bucket alignment, lowering linkage AUC (Table~\ref{tab:salt-linkage}); and (ii) it discards stable n-gram bucket identities accumulated within a window, so longitudinal aggregation across a rotation boundary loses concept continuity. Because the trained classifier is per training cycle, \emph{within-window} utility is unaffected by cadence as long as training and serving use the same salt — rotation events only matter at episode boundaries. We therefore recommend rotating once per training cycle as the default cadence, since (a) faster rotation requires re-training and re-deployment with no measured linkage benefit beyond Table~\ref{tab:salt-linkage}, and (b) slower rotation widens the linkage attack window. Per-session rotation, while maximally unlinkable, would force a fresh model per episode and is therefore impractical for any longitudinal screening use-case; deployments preferring stronger unlinkability should instead enable shuffle amplification (Sec.~III-E of the main paper).

\section{Local DP Mechanism and Composition}
\label{app:ldp}

\subsection{Mechanism and unbiased estimator}
Coordinate‑wise randomized response with privacy budget $\epsilon$ flips binarized presence $X_j \in \{0,1\}$ with probability $1-p$, where $p=\frac{e^{\epsilon}}{1+e^{\epsilon}}$. The privatized bit is
\[
Y_j=\begin{cases}
X_j,& \text{w.p. } p,\\
1-X_j,& \text{w.p. } 1-p.
\end{cases}
\]
For a bag $\mathcal{B}$ (e.g., posts of a user), the unbiased estimator of the true presence rate $\pi_j{=}\mathbb{E}[X_j]$ is
\[
\hat{\pi}_j=\frac{1}{|\mathcal{B}|}\sum_{x'\in\mathcal{B}}Y_j,\quad
\tilde{\pi}_j=\frac{\hat{\pi}_j-(1-p)}{2p-1}.
\]
\textbf{Variance.} Since $\text{Var}(Y_j)=\pi_j(1-\pi_j)(2p-1)^2 + p(1-p)$, we have
\[
\text{Var}(\tilde{\pi}_j)=\frac{\pi_j(1-\pi_j)}{|\mathcal{B}|} + \frac{p(1-p)}{(2p-1)^2 |\mathcal{B}|}.
\]
The second term is the privacy‑induced variance; it shrinks with larger bags or larger $\epsilon$.

\subsection{User‑level composition (bounded contributions)}
\textbf{Note.} The main paper adopts a one-shot per-user mechanism (ULDP-SSR, Sec.~III-E), yielding $\epsilon_{\text{hash}}{=}r\epsilon_b$ directly. Under independent per‑post RR at $\epsilon$ with a cap of $B$ posts, the conservative user‑level bound is $\epsilon_{\text{user}} \le B\epsilon$, which grows linearly in the number of posts. In practice, aggregation and binarization reduce attackability (see MIA/linkage). We report sensitivity to the cap $B$ at $\epsilon{=}2$ in Table~\ref{tab:comp-B}.

\subsection{Tighter accounting via R\'enyi DP}
\label{app:rdp}
The basic-composition bound used in the main text ($\epsilon_{\text{hash}}{=}r\epsilon_b$, $\epsilon_{\text{concept}}{\le}L\epsilon_c$) is parameter-free and convenient for privacy cards. R\'enyi DP (RDP) yields a tighter bound when an auxiliary $(\alpha,\delta)$ is acceptable. For the bit-level RR mechanism, the per-coordinate RDP at order $\alpha$ is
\[
\rho_\alpha^{\text{RR}}(\epsilon_b)=\tfrac{1}{\alpha-1}\ln\!\bigl((1-p_b)e^{\alpha\epsilon_b}+p_b e^{-\alpha\epsilon_b}\bigr),
\quad p_b=\tfrac{e^{\epsilon_b}}{1+e^{\epsilon_b}}.
\]
For $G$-ary KRR with $p_c{=}\tfrac{e^{\epsilon_c}}{e^{\epsilon_c}+G-1}$, $q_c{=}\tfrac{1}{e^{\epsilon_c}+G-1}$:
\[
\rho_\alpha^{\text{KRR}}(\epsilon_c)=\tfrac{1}{\alpha-1}\ln\!\bigl(q_c e^{\alpha\epsilon_c}+p_c e^{-\alpha\epsilon_c}+(G{-}2)q_c\bigr).
\]
The joint mechanism is $\bigl(r\rho_\alpha^{\text{RR}}{+}L\rho_\alpha^{\text{KRR}}\bigr)$-RDP; converting to $(\epsilon,\delta)$-DP adds $\tfrac{\ln(1/\delta)}{\alpha-1}$. Numerical values at default parameters $(r{=}64,\epsilon_b{=}0.1,L{=}48,\epsilon_c{=}0.05,\delta{=}10^{-6})$:

At $\alpha{=}8$, the user-level RDP bound ($\epsilon_{\text{user}}^{\text{RDP}}{\approx}4.3$) is roughly $2{\times}$ tighter than basic composition ($8.8$). The concept channel alone tightens at higher $\alpha$ (e.g.\ $0.95$ at $\alpha{=}16$, vs.\ $2.40$ basic) but the user budget is dominated by the hashed channel, whose optimum is near $\alpha{=}8$. The main text retains $r\epsilon_b{+}L\epsilon_c$ as a parameter-free upper bound.

\section{Additional Pseudocode}
\label{app:pseudocode}

\subsection{Client-side RR}
\begin{algorithm}[h]
\caption{Per-post randomized response (client)}
\label{alg:rr}
\begin{algorithmic}[1]
\Require Binarized hashed vector $X \in \{0,1\}^m$, $\epsilon$
\State $p \gets \frac{e^\epsilon}{1+e^\epsilon}$
\For{$j \in \{1,\dots,m\}$}
  \State $u \sim \text{Uniform}(0,1)$
  \If{$u < p$} \State $Y_j \gets X_j$
  \Else \State $Y_j \gets 1 - X_j$
  \EndIf
\EndFor
\State \Return $Y$
\end{algorithmic}
\end{algorithm}

\subsection{User aggregation (server) with de-biasing}
\begin{algorithm}[h]
\caption{Privatized user aggregation and de-biasing (server)}
\label{alg:agg}
\begin{algorithmic}[1]
\Require $\{Y^{(t)}\}_{t=1}^{T}$, $\epsilon$
\State $p \gets \frac{e^\epsilon}{1+e^\epsilon}$; $\hat{\pi} \gets \frac{1}{T}\sum_t Y^{(t)}$
\State $\tilde{\pi} \gets \frac{\hat{\pi} - (1-p)}{2p-1}$
\State \Return $\tilde{\pi}$
\end{algorithmic}
\end{algorithm}

\subsection{Final scoring and calibration}
\begin{algorithm}[h]
\caption{User-level scoring and isotonic calibration}
\label{alg:score}
\begin{algorithmic}[1]
\Require User vector $v_u=[\tilde{z}_u \| s_u]$, LR weights $(w,b)$, calibration map $\kappa$
\State $\hat{p}_u \gets \sigma(w^\top v_u + b)$
\State $\tilde{p}_u \gets \kappa(\hat{p}_u)$
\State \Return $\tilde{p}_u$
\end{algorithmic}
\end{algorithm}
